7 # NeXus data as Python trees
8 # ==========================
9 # The `nexus.tree` modules are designed to accomplish two goals:
10 #
11 # 1. To provide convenient access to existing data contained in NeXus files.
12 # 2. To enable new NeXus data to be created and manipulated interactively.
13 #
14 # These goals are achieved by mapping hierarchical NeXus data structures directly
15 # into python objects, which either represent NeXus groups or NeXus fields.
16 # Entries in a group are referenced much like fields in a class are referenced in
17 # python. The entire data hierarchy can be referenced at any time, whether the
18 # NeXus data has been loaded in from an existing NeXus file or created dynamically
19 # within the python session. This provides a much more natural scripting interface
20 # to NeXus data than the directory model of the `nexus.napi` interface.
21 #
22 # Example 1: Loading a NeXus file
23 # -------------------------------
24 # The following commands loads NeXus data from a file, displays (some of) the
25 # contents as a tree, and then accesses individual data items.
26 #
27 # >>> from nexpy.api import nexus as nx
28 # >>> a=nx.load('sns/data/ARCS_7326.nxs')
29 # >>> print a.tree
30 # root:NXroot
31 # @HDF5_Version = 1.8.2
32 # @NeXus_version = 4.2.1
33 # @file_name = ARCS_7326.nxs
34 # @file_time = 2010-05-05T01:59:25-05:00
35 # entry:NXentry
36 # data:NXdata
37 # data = float32(631x461x4x825)
38 # @axes = rotation_angle:tilt_angle:sample_angle:time_of_flight
39 # @signal = 1
40 # rotation_angle = float32(632)
41 # @units = degree
42 # sample_angle = [ 210. 215. 220. 225. 230.]
43 # @units = degree
44 # tilt_angle = float32(462)
45 # @units = degree
46 # time_of_flight = float32(826)
47 # @units = microsecond
48 # run_number = 7326
49 # sample:NXsample
50 # pulse_time = 2854.94747365
51 # @units = microsecond
52 # .
53 # .
54 # .
55 # >>> a.entry.run_number
56 # NXfield(7326)
57 #
58 # So the tree returned from load() has an entry for each group, field and
59 # attribute. You can traverse the hierarchy using the names of the groups. For
60 # example, tree.entry.instrument.detector.distance is an example of a field
61 # containing the distance to each pixel in the detector. Entries can also be
62 # referenced by NXclass name, such as tree.NXentry[0].instrument. Since there may
63 # be multiple entries of the same NeXus class, the NXclass attribute returns a
64 # (possibly empty) list.
65 #
66 # The load() and save() functions are implemented using the class
67 # `nexus.tree.NeXusTree`, a subclass of `nexus.napi.NeXus` which allows all the
68 # usual API functions.
69 #
70 # Example 2: Creating a NeXus file dynamically
71 # --------------------------------------------
72 # The second example shows how to create NeXus data dynamically and saves it to a
73 # file. The data are first created as Numpy arrays
74 #
75 # >>> import numpy as np
76 # >>> x=y=np.linspace(0,2*np.pi,101)
77 # >>> X,Y=np.meshgrid(y,x)
78 # >>> z=np.sin(X)*np.sin(Y)
79 #
80 # Then a NeXus data groups are created and the data inserted to produce a
81 # NeXus-compliant structure that can be saved to a file.
82 #
83 # >>> root=nx.NXroot(NXentry())
84 # >>> print root.tree
85 # root:NXroot
86 # entry:NXentry
87 # >>> root.entry.data=nx.NXdata(z,[x,y])
88 #
89 # Additional metadata can be inserted before saving the data to a file.
90 #
91 # >>> root.entry.sample=nx.NXsample()
92 # >>> root.entry.sample.temperature = 40.0
93 # >>> root.entry.sample.temperature.units = 'K'
94 # >>> root.save('example.nxs')
95 #
96 # NXfield objects have much of the functionality of Numpy arrays. They may be used
97 # in simple arithmetic expressions with other NXfields, Numpy arrays or scalar
98 # values and will be cast as ndarray objects if used as arguments in Numpy
99 # modules.
100 #
101 # >>> x=nx.NXfield(np.linspace(0,10.0,11))
102 # >>> x
103 # NXfield([ 0. 1. 2. ..., 8. 9. 10.])
104 # >>> x + 10
105 # NXfield([ 10. 11. 12. ..., 18. 19. 20.])
106 # >>> sin(x)
107 # array([ 0. , 0.84147098, 0.90929743, ..., 0.98935825,
108 # 0.41211849, -0.54402111])
109 #
110 # If the arithmetic operation is assigned to a NeXus group attribute, it will be
111 # automatically cast as a valid NXfield object with the type and shape determined
112 # by the Numpy array type and shape.
113 #
114 # >>> entry.data.result = sin(x)
115 # >>> entry.data.result
116 # NXfield([ 0. 0.84147098 0.90929743 ..., 0.98935825 0.41211849
117 # -0.54402111])
118 # >>> entry.data.result.dtype, entry.data.result.shape
119 # (dtype('float64'), (11,))
120 #
121 # NeXus Objects
122 # -------------
123 # Properties of the entry in the tree are referenced by attributes that depend
124 # on the object type, different nx attributes may be available.
125 #
126 # Objects (class NXobject) have attributes shared by both groups and fields::
127 # * nxname object name
128 # * nxclass object class for groups, 'NXfield' for fields
129 # * nxgroup group containing the entry, or None for the root
130 # * attrs dictionary of NeXus attributes for the object
131 #
132 # Groups (class NXgroup) have attributes for accessing children::
133 # * entries dictionary of entries within the group
134 # * component('nxclass') return group entries of a particular class
135 # * dir() print the list of entries in the group
136 # * tree return the list of entries and subentries in the group
137 # * plot() plot signal and axes for the group, if available
138 #
139 # Fields (class NXfield) have attributes for accessing data:
140 # * shape dimensions of data in the field
141 # * dtype data type
142 # * nxdata data in the field
143 #
144 # Linked fields or groups (class NXlink) have attributes for accessing the link::
145 # * nxlink reference to the linked field or group
146 #
147 # NeXus attributes (class NXattr) have a type and a value only::
148 # * dtype attribute type
149 # * nxdata attribute data
150 #
151 # There is a subclass of NXgroup for each group class defined by the NeXus standard,
152 # so it is possible to create an NXgroup of NeXus class NXsample directly using:
153 #
154 # >>> sample = NXsample()
155 #
156 # The default group name will be the class name following the 'NX', so the above
157 # group will have an nxname of 'sample'. However, this is overridden by the
158 # attribute name when it is assigned as a group attribute, e.g.,
159 #
160 # >>> entry.sample1 = NXsample()
161 # >>> entry.sample1.nxname
162 # sample1
163 #
164 # You can traverse the tree by component class instead of component name. Since
165 # there may be multiple components of the same class in one group you will need to
166 # specify which one to use. For example,
167 #
168 # tree.NXentry[0].NXinstrument[0].NXdetector[0].distance
169 #
170 # references the first detector of the first instrument of the first entry.
171 # Unfortunately, there is no guarantee regarding the order of the entries, and it
172 # may vary from call to call, so this is mainly useful in iterative searches.
173 #
174 #
175 # Unit Conversion
176 # ---------------
177 # Data can be stored in the NeXus file in a variety of units, depending on which
178 # facility is storing the file. This makes life difficult for reduction and
179 # analysis programs which must know the units they are working with. Our solution
180 # to this problem is to allow the reader to retrieve data from the file in
181 # particular units. For example, if detector distance is stored in the file using
182 # millimeters you can retrieve them in meters using::
183 #
184 # entry.instrument.detector.distance.convert('m')
185 #
186 # See `nexus.unit` for more details on the unit formats supported.
187 #
188 # Reading and Writing Slabs
189 # -------------------------
190 # The slab interface to field data works by opening the file handle and keeping it
191 # open as long as the slab interface is needed. This is done in python 2.5 using
192 # the with statement. Once the context is entered, get() and put() methods on the
193 # object allow you to read and write data a slab at a time. For example::
194 #
195 # # Read a Ni x Nj x Nk array one vector at a time
196 # with root.NXentry[0].data.data as slab:
197 # Ni,Nj,Nk = slab.shape
198 # size = [1,1,Nk]
199 # for i in range(Ni):
200 # for j in range(Nj):
201 # value = slab.get([i,j,0],size)
202 #
203 # The equivalent can be done in Python 2.4 and lower using the context
204 # functions __enter__ and __exit__::
205 #
206 # slab = data.slab.__enter__()
207 # ... do the slab functions ...
208 # data.slab.__exit__()
209 #
210 # Plotting NeXus data
211 # -------------------
212 # There is a plot() method for groups that automatically looks for 'signal' and
213 # 'axes' attributes within the group in order to determine what to plot. These are
214 # defined by the 'nxsignal' and 'nxaxes' properties of the group. This means that
215 # the method will determine whether the plot should be one- or two- dimensional.
216 # For higher than two dimensions, only the top slice is plotted by default.
217 #
218 # The plot method accepts as arguments the standard matplotlib.pyplot.plot format
219 # strings to customize one-dimensional plots, axis and scale limits, and will
220 # transmit keyword arguments to the matplotlib plotting methods.
221 #
222 # >>> a=nx.load('chopper.nxs')
223 # >>> a.entry.monitor1.plot()
224 # >>> a.entry.monitor2.plot('r+', xmax=2600)
225 #
226 # It is possible to plot over the existing figure with the oplot() method and to
227 # plot with logarithmic intensity scales with the logplot() method. The x- and
228 # y-axes can also be rendered logarithmically using the logx and logy keywards.
229 #
230 # Although the plot() method uses matplotlib by default to plot the data, you can replace
231 # this with your own plotter by setting nexus.NXgroup._plotter to your own plotter
232 # class. The plotter class has one method::
233 #
234 # plot(signal, axes, entry, title, format, **opts)
235 #
236 # where signal is the field containing the data, axes are the fields listing the
237 # signal sample points, entry is file/path within the file to the data group and
238 # title is the title of the group or the parent NXentry, if available.
239 #

270 # Structure-based interface to the NeXus file API.
271 #
272 # Usage::
273 #
274 # file = NeXusTree(filename, ['r','rw','w'])
275 # - open the NeXus file
276 # root = file.readfile()
277 # - read the structure of the NeXus file. This returns a NeXus tree.
278 # file.writefile(root)
279 # - write a NeXus tree to the file.
280 # data = file.readpath(path)
281 # - read data from a particular path
282 #
283 # Example::
284 #
285 # nx = NeXusTree('REF_L_1346.nxs','r')
286 # tree = nx.readfile()
287 # for entry in tree.NXentry:
288 # process(entry)
289 # copy = NeXusTree('modified.nxs','w')
290 # copy.writefile(tree)
291 #
292 # Note that the large datasets are not loaded immediately. Instead, the
293 # when the data set is requested, the file is reopened, the data read, and
294 # the file closed again. open/close are available for when we want to
295 # read/write slabs without the overhead of moving the file cursor each time.
296 # The NXdata objects in the returned tree hold the object values.

574 #
575 # Class for NeXus attributes of a NXfield or NXgroup object.
576 #
577 # This class is only used for NeXus attributes that are stored in a
578 # NeXus file and helps to distinguish them from Python attributes.
579 # There are two Python attributes for each NeXus attribute.
580 #
581 # Python Attributes
582 # -----------------
583 # nxdata : string, Numpy scalar, or Numpy ndarray
584 # The value of the NeXus attribute.
585 # dtype : string
586 # The data type of the NeXus attribute. This is set to 'char' for
587 # a string attribute or the string of the corresponding Numpy data type
588 # for a numeric attribute.
589 #
590 # NeXus Attributes
591 # ----------------
592 # NeXus attributes are stored in the 'attrs' dictionary of the parent object,
593 # NXfield or NXgroup, but can often be referenced or assigned using the
594 # attribute name as if it were an object attribute.
595 #
596 # For example, after assigning the NXfield, the following three attribute
597 # assignments are all equivalent::
598 #
599 # >>> entry.sample.temperature = NXfield(40.0)
600 # >>> entry.sample.temperature.attrs['units'] = 'K'
601 # >>> entry.sample.temperature.units = NXattr('K')
602 # >>> entry.sample.temperature.units = 'K'
603 #
604 # The third version above is only allowed for NXfield attributes and is
605 # not allowed if the attribute has the same name as one of the following
606 # internally defined attributes, i.e.,
607 #
608 # ['entries', 'attrs', 'dtype','shape']
609 #
610 # or if the attribute name begins with 'nx' or '_'. It is only possible to
611 # reference attributes with one of the proscribed names using the 'attrs'
612 # dictionary.
613 #
614 #
615 class NXattr(object):
616 
617 
618  def __init__(self,value=None,dtype=''):

620  self._data,self._dtype = value.nxdata,value.dtype

1085 # NeXus Attributes
1086 # ----------------
1087 # NeXus attributes are stored in the 'attrs' dictionary of the NXfield, but
1088 # can usually be assigned or referenced as if they are Python attributes, as
1089 # long as the attribute name is not the same as one of those listed above.
1090 # This is to simplify typing in an interactive session and should not cause
1091 # any problems because there is no name clash with attributes so far defined
1092 # within the NeXus standard. When writing modules, it is recommended that the
1093 # attributes always be referenced using the 'attrs' dictionary if there is
1094 # any doubt.
1095 #
1096 # a) Assigning a NeXus attribute
1097 #
1098 # In the example below, after assigning the NXfield, the following three
1099 # NeXus attribute assignments are all equivalent:
1100 #
1101 # >>> entry.sample.temperature = NXfield(40.0)
1102 # >>> entry.sample.temperature.attrs['units'] = 'K'
1103 # >>> entry.sample.temperature.units = NXattr('K')
1104 # >>> entry.sample.temperature.units = 'K'
1105 #
1106 # b) Referencing a NeXus attribute
1107 #
1108 # If the name of the NeXus attribute is not the same as any of the Python
1109 # attributes listed above, or one of the methods listed below, or any of the
1110 # attributes defined for Numpy arrays, they can be referenced as if they were
1111 # a Python attribute of the NXfield. However, it is only possible to reference
1112 # attributes with one of the proscribed names using the 'attrs' dictionary.
1113 #
1114 # >>> entry.sample.temperature.tree = 10.0
1115 # >>> entry.sample.temperature.tree
1116 # temperature = 40.0
1117 # @tree = 10.0
1118 # @units = K
1119 # >>> entry.sample.temperature.attrs['tree']
1120 # NXattr(10.0)

1122 # Numerical Operations on NXfields
1123 # --------------------------------
1124 # NXfields usually consist of arrays of numeric data with associated
1125 # meta-data, the NeXus attributes. The exception is when they contain
1126 # character strings. This makes them similar to Numpy arrays, and this module
1127 # allows the use of NXfields in numerical operations in the same way as Numpy
1128 # ndarrays. NXfields are technically not a sub-class of the ndarray class, but
1129 # most Numpy operations work on NXfields, returning either another NXfield or,
1130 # in some cases, an ndarray that can easily be converted to an NXfield.
1131 #
1132 # >>> x = NXfield((1.0,2.0,3.0,4.0))
1133 # >>> print x+1
1134 # [ 2. 3. 4. 5.]
1135 # >>> print 2*x
1136 # [ 2. 4. 6. 8.]
1137 # >>> print x/2
1138 # [ 0.5 1. 1.5 2. ]
1139 # >>> print x**2
1140 # [ 1. 4. 9. 16.]
1141 # >>> print x.reshape((2,2))
1142 # [[ 1. 2.]
1143 # [ 3. 4.]]
1144 # >>> y = NXfield((0.5,1.5,2.5,3.5))
1145 # >>> x+y
1146 # NXfield(name=x,value=[ 1.5 3.5 5.5 7.5])
1147 # >>> x*y
1148 # NXfield(name=x,value=[ 0.5 3. 7.5 14. ])
1149 # >>> (x+y).shape
1150 # (4,)
1151 # >>> (x+y).dtype
1152 # dtype('float64')
1153 #
1154 # All these operations return valid NXfield objects containing the same
1155 # attributes as the first NXobject in the expression. The 'reshape' and
1156 # 'transpose' methods also return NXfield objects.
1157 #
1158 # It is possible to use the standard slice syntax.
1159 #
1160 # >>> x=NXfield(np.linspace(0,10,11))
1161 # >>> x
1162 # NXfield([ 0. 1. 2. ..., 8. 9. 10.])
1163 # >>> x[2:5]
1164 # NXfield([ 2. 3. 4.])
1165 #
1166 # In addition, it is possible to use floating point numbers as the slice
1167 # indices. If one of the indices is not integer, both indices are used to
1168 # extract elements in the array with values between the two index values.
1169 #
1170 # >>> x=NXfield(np.linspace(0,100.,11))
1171 # >>> x
1172 # NXfield([ 0. 10. 20. ..., 80. 90. 100.])
1173 # >>> x[20.:50.]
1174 # NXfield([ 20. 30. 40. 50.])
1175 #
1176 # The standard Numpy ndarray attributes and methods will also work with
1177 # NXfields, but will return scalars or Numpy arrays.
1178 #
1179 # >>> x.size
1180 # 4
1181 # >>> x.sum()
1182 # 10.0
1183 # >>> x.max()
1184 # 4.0
1185 # >>> x.mean()
1186 # 2.5
1187 # >>> x.var()
1188 # 1.25
1189 # >>> x.reshape((2,2)).sum(1)
1190 # array([ 3., 7.])
1191 #
1192 # Finally, NXfields are cast as ndarrays for operations that require them.
1193 # The returned value will be the same as for the equivalent ndarray
1194 # operation, e.g.,
1195 #
1196 # >>> np.sin(x)
1197 # array([ 0.84147098, 0.90929743, 0.14112001, -0.7568025 ])
1198 # >>> np.sqrt(x)
1199 # array([ 1. , 1.41421356, 1.73205081, 2. ])

1979 # path : string

1981 # using the convention for unix file paths.
1982 # group : NXobject (NXgroup or subclass of NXgroup)

1989 # nxclass : string

2013 # NeXus Group Entries
2014 # -------------------
2015 # Just as in a NeXus file, NeXus groups can contain either data or other
2016 # groups, represented by NXfield and NXgroup objects respectively. To
2017 # distinguish them from regular Python attributes, all NeXus objects are
2018 # stored in the 'entries' dictionary of the NXgroup. However, they can usually
2019 # be assigned or referenced as if they are Python attributes, i.e., using the
2020 # dictionary name directly as the group attribute name, as long as this name
2021 # is not the same as one of the Python attributes defined above or as one of
2022 # the NXfield Python attributes.

2024 # a) Assigning a NeXus object to a NeXus group
2025 #
2026 # In the example below, after assigning the NXgroup, the following three
2027 # NeXus object assignments to entry.sample are all equivalent:
2028 #
2029 # >>> entry.sample = NXsample()
2030 # >>> entry.sample['temperature'] = NXfield(40.0)
2031 # >>> entry.sample.temperature = NXfield(40.0)
2032 # >>> entry.sample.temperature = 40.0
2033 # >>> entry.sample.temperature
2034 # NXfield(40.0)

2048 # If the name of the NeXus object is not the same as any of the Python
2049 # attributes listed above, or the methods listed below, they can be referenced

2052 # dictionary, i.e.,
2053 #

2064 # NeXus Attributes
