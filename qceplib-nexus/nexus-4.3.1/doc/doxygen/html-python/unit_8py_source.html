5 # Define unit conversion support for NeXus style units.
6 #
7 # The unit format is somewhat complicated. There are variant spellings
8 # and incorrect capitalization to worry about, as well as forms such as
9 # "mili*metre" and "1e-7 seconds".
10 #
11 # This is a minimal implementation of units including only what I happen to
12 # need now. It does not support the complete dimensional analysis provided
13 # by the package udunits on which NeXus is based, or even the units used
14 # in the NeXus definition files.
15 #
16 # Unlike other units modules, this module does not carry the units along
17 # with the value, but merely provides a conversion function for
18 # transforming values.
19 #
20 # Usage example::
21 #
22 # import nxs.unit
23 # u = nxs.unit.Converter('mili*metre') # Units stored in mm
24 # v = u(3000,'m') # Convert the value 3000 mm into meters
25 #
26 # NeXus example::
27 #
28 # # Load sample orientation in radians regardless of how it is stored.
29 # # 1. Open the path
30 # file.openpath('/entry1/sample/sample_orientation')
31 # # 2. scan the attributes, retrieving 'units'
32 # units = [for attr,value in file.attrs() if attr == 'units']
33 # # 3. set up the converter (assumes that units actually exists)
34 # u = nxs.unit.Converter(units[0])
35 # # 4. read the data and convert to the correct units
36 # v = u(file.read(),'radians')
37 #
38 # This is a standalone module, not relying on either DANSE or NeXus, and
39 # can be used for other unit conversion tasks.
40 #
41 # Note: minutes are used for angle and seconds are used for time. We
42 # cannot tell what the correct interpretation is without knowing something
43 # about the fields themselves. If this becomes an issue, we will need to
44 # allow the application to set the dimension for the units rather than
45 # getting the dimension from the units as we are currently doing.

79 def _build_metric_units(unit,abbr):
80  prefix = dict(peta=1e15,tera=1e12,giga=1e9,mega=1e6,kilo=1e3,
81  deci=1e-1,centi=1e-2,milli=1e-3,mili=1e-3,micro=1e-6,
82  nano=1e-9,pico=1e-12,femto=1e-15)
83  short_prefix = dict(P=1e15,T=1e12,G=1e9,M=1e6,k=1e3,
84  d=1e-1,c=1e-2,m=1e-3,u=1e-6,
85  n=1e-9,p=1e-12,f=1e-15)
86  map = {abbr:1}
87  map.update([(P+abbr,scale) for (P,scale) in short_prefix.iteritems()])
88  for name in [unit,unit.capitalize()]:
89  map.update({name:1,name+'s':1})
90  map.update([(P+name,scale) for (P,scale) in prefix.iteritems()])
91  map.update([(P+'*'+name,scale) for (P,scale) in prefix.iteritems()])
92  map.update([(P+name+'s',scale) for (P,scale) in prefix.iteritems()])
93  return map

99 def _build_plural_units(**kw):
100  map = {}
101  map.update([(name,scale) for name,scale in kw.iteritems()])
102  map.update([(name+'s',scale) for name,scale in kw.iteritems()])
103  return map

153 class Converter(object):
154  # Define the units, using both American and European spelling.
155  scalemap = None
156  scalebase = 1
157  dims = _build_all_units()
158 
159  def __init__(self,name):
160  self.base = name
161  for map in self.dims:
162  if name in map:
163  self.scalemap = map
164  self.scalebase = self.scalemap[name]
165  break
166  else:
167  self.scalemap = {'': 1}
168  self.scalebase = 1
169  #raise ValueError, "Unknown unit %s"%name
170 
171  def scale(self, units=""):
172  if units == "" or self.scalemap is None: return 1
173  return self.scalebase/self.scalemap[units]
174 
175  def __call__(self, value, units=""):
176  # Note: calculating a*1 rather than simply returning a would produce
177  # an unnecessary copy of the array, which in the case of the raw
178  # counts array would be bad. Sometimes copying and other times
179  # not copying is also bad, but copy on modify semantics isn't
180  # supported.
181  if units == "" or self.scalemap is None: return value
182  try:
183  return value * (self.scalebase/self.scalemap[units])
184  except KeyError:
185  raise KeyError("%s not in %s"%(units," ".join(self.scalemap.keys())))
